# Git Commands to Run (Copy and paste these one by one)

# 1. Initialize repository
git init

# 2. Add all files
git add .

# 3. Commit
git commit -m "Restore to simple localStorage version - removed Firebase dependencies"

# 4. Add remote (your GitHub repo)
git remote add origin https://github.com/S0N59/tgapp.git

# 5. Set branch to main
git branch -M main

# 6. Push to GitHub
git push -u origin main

# If repository doesn't exist, create it first at: https://github.com/new
